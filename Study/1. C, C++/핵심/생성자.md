## 복사 생성자

```cpp
#include <iostream>
#include <string>
using namespace std;

class Test {
	int value;
	
public:
	Test(int value);
	Test(const Test& other);
}

Test::Test(int value) {
	cout << "기본 생성자" << endl;
}

Test::Test(const Test& other) {
	cout << "복사 생성자" << endl;
}

int main() {
	Test test1(1); // 생성자 호출
	Test test2(test1); // 복사 생성자 호출
	Test test3 = test2; // 복사 생성자 호출
}
```
